import 'dart:io';

List<int> takeInputFromUser() {
  stdout.write("Please enter the number of elements you wish to enter: ");
  int? numOfElements = int.parse(stdin.readLineSync()!);
  List<int> userInput = [];
  for (int i = 0; i < numOfElements; i++) {
    stdout.write("Please enter element ${i + 1}: ");
    userInput.add(int.parse(stdin.readLineSync()!));
  }
  return userInput;
}

void printList(List list) {
  for (int i = 0; i < list.length; i++) {
    stdout.write('Element ${i + 1}: ${list[i]}\t');
  }
  print('');
}

int getSumOfElements(List<int> list) {
  int sum = 0;
  for (int i = 0; i < list.length; i++) {
    sum += list[i];
  }
  return sum;
}

double getAverageOfList(List<int> list) {
  int sum = getSumOfElements(list);
  return sum / list.length;
}

main() {
  List<int> userInput = takeInputFromUser();
  printList(userInput);
  print('Sum of elements: ${getSumOfElements(userInput)}');
  print('List Average:  ${getAverageOfList(userInput)}');
}
